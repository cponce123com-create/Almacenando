MEJORA 1 — Corregir el formato del PDF: firma al fondo y numeración reiniciada
En artifacts/legado/src/pages/legacy/form.tsx, en la función convertMarkdownToHtml, corregir el problema de numeración de listas. El problema es que las listas numeradas de distintas cláusulas se tratan como una sola lista continua y los números no reinician.
Reemplazar la línea que maneja los ítems numerados por esta lógica que convierte cada grupo separado de items en su propia lista <ol> independiente:
typescriptfunction convertMarkdownToHtml(text: string): string {
  const lines = text.split("\n");
  let html = "";
  let inOrderedList = false;
  let inUnorderedList = false;

  for (const line of lines) {
    // Títulos
    if (/^# (.+)$/.test(line)) {
      if (inOrderedList) { html += "</ol>"; inOrderedList = false; }
      if (inUnorderedList) { html += "</ul>"; inUnorderedList = false; }
      html += line.replace(/^# (.+)$/, '<h1 style="font-size:20px;font-weight:bold;text-align:center;margin:0 0 6px;letter-spacing:1px;">$1</h1>');
    } else if (/^## (.+)$/.test(line)) {
      if (inOrderedList) { html += "</ol>"; inOrderedList = false; }
      if (inUnorderedList) { html += "</ul>"; inUnorderedList = false; }
      html += line.replace(/^## (.+)$/, '<h2 style="font-size:14px;font-weight:bold;margin:20px 0 6px;border-bottom:1px solid #ccc;padding-bottom:3px;">$1</h2>');
    } else if (/^### (.+)$/.test(line)) {
      if (inOrderedList) { html += "</ol>"; inOrderedList = false; }
      if (inUnorderedList) { html += "</ul>"; inUnorderedList = false; }
      html += line.replace(/^### (.+)$/, '<h3 style="font-size:13px;font-weight:bold;margin:14px 0 4px;">$1</h3>');
    // Separador ---
    } else if (/^---$/.test(line)) {
      if (inOrderedList) { html += "</ol>"; inOrderedList = false; }
      if (inUnorderedList) { html += "</ul>"; inUnorderedList = false; }
      html += '<hr style="border:none;border-top:1px solid #ccc;margin:14px 0;" />';
    // Lista numerada — reinicia por cada bloque separado
    } else if (/^\d+\. (.+)$/.test(line)) {
      if (inUnorderedList) { html += "</ul>"; inUnorderedList = false; }
      if (!inOrderedList) {
        html += '<ol style="margin:8px 0;padding-left:28px;" start="1">';
        inOrderedList = true;
      }
      const content = line.replace(/^\d+\. (.+)$/, "$1").replace(/\*\*(.+?)\*\*/g, "<strong>$1</strong>").replace(/\*(.+?)\*/g, "<em>$1</em>");
      html += `<li style="margin:5px 0;line-height:1.7;">${content}</li>`;
    // Lista con guión
    } else if (/^- (.+)$/.test(line)) {
      if (inOrderedList) { html += "</ol>"; inOrderedList = false; }
      if (!inUnorderedList) { html += '<ul style="margin:8px 0;padding-left:28px;">'; inUnorderedList = true; }
      const content = line.replace(/^- (.+)$/, "$1").replace(/\*\*(.+?)\*\*/g, "<strong>$1</strong>").replace(/\*(.+?)\*/g, "<em>$1</em>");
      html += `<li style="margin:5px 0;line-height:1.7;">${content}</li>`;
    // Blockquote
    } else if (/^> (.+)$/.test(line)) {
      if (inOrderedList) { html += "</ol>"; inOrderedList = false; }
      if (inUnorderedList) { html += "</ul>"; inUnorderedList = false; }
      const content = line.replace(/^> (.+)$/, "$1").replace(/\*\*(.+?)\*\*/g, "<strong>$1</strong>").replace(/\*(.+?)\*/g, "<em>$1</em>");
      html += `<blockquote style="border-left:3px solid #999;padding:6px 14px;margin:10px 0;color:#444;font-style:italic;background:#f9f9f9;">${content}</blockquote>`;
    // Línea vacía — cierra listas abiertas
    } else if (line.trim() === "") {
      if (inOrderedList) { html += "</ol>"; inOrderedList = false; }
      if (inUnorderedList) { html += "</ul>"; inUnorderedList = false; }
      html += '<div style="margin:6px 0;"></div>';
    // Párrafo normal
    } else {
      if (inOrderedList) { html += "</ol>"; inOrderedList = false; }
      if (inUnorderedList) { html += "</ul>"; inUnorderedList = false; }
      const content = line.replace(/\*\*(.+?)\*\*/g, "<strong>$1</strong>").replace(/\*(.+?)\*/g, "<em>$1</em>");
      html += `<p style="margin:7px 0;line-height:1.8;">${content}</p>`;
    }
  }

  if (inOrderedList) html += "</ol>";
  if (inUnorderedList) html += "</ul>";
  return html;
}
