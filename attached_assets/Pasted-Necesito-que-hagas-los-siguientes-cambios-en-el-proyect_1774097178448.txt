CAMBIO 5 — Página pública para contactos de confianza
Primero agregar en el backend el endpoint que falta.
Archivo: artifacts/api-server/src/routes/activation.ts
Agregar este endpoint GET antes del export default:
typescriptrouter.get("/trusted-contact-info/:token", async (req, res) => {
  const contact = await db
    .select()
    .from(trustedContactsTable)
    .where(eq(trustedContactsTable.inviteToken, req.params.token))
    .limit(1);

  if (contact.length === 0) {
    res.status(404).json({ error: "Not found", message: "Token inválido o expirado" });
    return;
  }

  const trustedContact = contact[0];

  if (trustedContact.inviteTokenExpiresAt && trustedContact.inviteTokenExpiresAt < new Date()) {
    res.status(404).json({ error: "Expired", message: "Este enlace ha expirado" });
    return;
  }

  const profile = await db
    .select()
    .from(profilesTable)
    .where(eq(profilesTable.userId, trustedContact.userId))
    .limit(1);

  res.json({
    contactName: trustedContact.fullName,
    ownerName: profile[0]?.fullName || "Usuario de Legado",
    userId: trustedContact.userId,
  });
});
```

Asegurarse de importar `profilesTable` desde `@workspace/db` si no está ya importado.
```
