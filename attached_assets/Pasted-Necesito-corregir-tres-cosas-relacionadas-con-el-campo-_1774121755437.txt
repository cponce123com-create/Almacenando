CAMBIO 4 — Verificar el endpoint backend
Archivo: artifacts/api-server/src/routes/trusted-contacts.ts
Verificar que existe el endpoint POST /verify-dni y que:

typescriptrouter.post("/verify-dni", requireAuth, async (req, res) => {
  const { dni } = req.body;

  if (!dni || dni.replace(/\D/g, "").length !== 8) {
    res.status(400).json({ error: "El DNI debe tener exactamente 8 dígitos" });
    return;
  }

  const token = process.env.RENIEC_API_TOKEN;
  if (!token) {
    res.status(503).json({ error: "Verificación de DNI no disponible" });
    return;
  }

  try {
    const response = await fetch(
      `https://api.decolecta.com/v1/reniec/dni?numero=${dni.trim()}`,
      {
        headers: {
          "Content-Type": "application/json",
          "Authorization": `Bearer ${token}`,
        },
      }
    );

    if (response.status === 404) {
      res.status(404).json({ error: "DNI no encontrado en RENIEC" });
      return;
    }

    if (!response.ok) {
      const errBody = await response.text();
      console.error("[RENIEC] error response:", response.status, errBody);
      res.status(500).json({ error: "Error al consultar RENIEC" });
      return;
    }

    const data = await response.json();

    res.json({
      valid: true,
      firstName: data.first_name,
      firstLastName: data.first_last_name,
      secondLastName: data.second_last_name,
      fullName: data.full_name,
      documentNumber: data.document_number,
    });
  } catch (err) {
    console.error("[RENIEC]", err);
    res.status(500).json({ error: "Error interno al verificar DNI" });
  }
});

Agregar un console.log al inicio del endpoint para confirmar que se está ejecutando:
